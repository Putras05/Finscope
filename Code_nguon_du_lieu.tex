\section{Giới thiệu}

\noindent FinScope sử dụng dữ liệu thời gian thực 53 mã chứng khoán
niêm yết trên Sở Giao dịch Chứng khoán Thành phố Hồ Chí Minh (HOSE)
thông qua thư viện mã nguồn mở \texttt{vnstock 4.0.4}. Toàn bộ logic
\textbf{lấy -- làm sạch -- chuẩn hoá} dữ liệu được tập trung tại
thư mục \texttt{data/} của mã nguồn. Tài liệu này trích đoạn ba file
cốt lõi của tầng dữ liệu để ban giám khảo đối chiếu:

\begin{itemize}
  \item \texttt{data/\_clients.py} -- singleton kết nối + bộ điều tiết
        tốc độ truy vấn (token bucket) dưới 20 yêu cầu/phút.
  \item \texttt{data/fetcher.py} -- truy xuất giá lịch sử 1 mã, kèm
        các chỉ báo kỹ thuật cơ bản (MA, RSI).
  \item \texttt{data/market.py} -- truy xuất snapshot thị trường (giá
        khớp lệnh hiện tại, vốn hoá, room nước ngoài) cho danh sách mã.
\end{itemize}

Toàn bộ phần còn lại của ứng dụng (14 trang giao diện, 8 mô hình dự
báo, 4 thuật toán toán -- tài chính cổ điển, hệ thống xác thực và cá
nhân hoá) chạy trực tiếp trên \textbf{Streamlit Community Cloud} --
ban giám khảo truy cập qua trình duyệt mà không cần cài đặt môi
trường Python (chi tiết tại tệp \texttt{Mo\_ta\_san\_pham.pdf}).

% =====================================================================
%  data/_clients.py
% =====================================================================
\section{data/\_clients.py -- Singleton client và token bucket}

\noindent File này giải quyết hai bài toán: (i) tái sử dụng kết nối
\texttt{vnstock} -- khởi tạo một lần, dùng chung mọi nơi (tránh
TLS handshake 200--500\,ms cho mỗi lần truy vấn); (ii) chia sẻ một
\emph{token bucket} duy nhất giữa mọi luồng (foreground + warmer
nền) để tổng tốc độ không vượt giới hạn của \texttt{vnstock}.

\VerbatimInput{data/_clients.py}

% =====================================================================
%  data/fetcher.py
% =====================================================================
\section{data/fetcher.py -- Truy xuất giá lịch sử và chỉ báo}

\noindent Hàm \texttt{fetch\_data(ticker)} là cửa ngõ chính để mọi tầng
phía trên (mô hình, biểu đồ, bộ máy tín hiệu) lấy chuỗi giá. Kết quả
được cache đĩa 24 giờ thông qua \texttt{@st.cache\_data(persist="disk")};
các chỉ báo cơ bản (MA5/20/50, MA5\_Vol, RSI14) được tính ngay tại tầng
dữ liệu để mọi page tái sử dụng.

\VerbatimInput{data/fetcher.py}

% =====================================================================
%  data/market.py
% =====================================================================
\section{data/market.py -- Snapshot toàn thị trường}

\noindent Hàm \texttt{market\_snapshot(tickers)} dùng cho các trang Tổng
quan thị trường, Phân tích cơ bản và Dashboard. Một lần truy vấn
\texttt{price\_board} của \texttt{vnstock} trả về dữ liệu cho cả danh
sách 53 mã, tiết kiệm thời gian so với truy vấn từng mã một.

\VerbatimInput{data/market.py}

\section{Ghi chú thực thi}

\noindent Hai chi tiết kỹ thuật đáng chú ý:

\begin{enumerate}
  \item \textbf{Bao \texttt{contextlib.redirect\_stdout/redirect\_stderr}}:
        \texttt{vnstock} in các thông báo bằng tiếng Việt ra console
        (cảnh báo cập nhật, quảng cáo bản Insiders). Trên hệ điều hành
        Windows với mã hoá \texttt{cp1252}, các ký tự tiếng Việt gây
        \texttt{UnicodeEncodeError} làm vỡ hàm \texttt{fetch}. Nhóm
        bọc các lời gọi \texttt{vnstock} trong khối nuốt stdout/stderr
        để tránh sự cố này (đã quan sát trên mã ACB).

  \item \textbf{Tier-aware throttle}: hàm \texttt{throttle()} phân biệt
        hai loại lời gọi -- foreground (\texttt{is\_foreground=True})
        chỉ chờ tối thiểu 1,6 giây; background/warmer chờ 3,4 giây và
        nhường nhịn khi foreground vừa claim slot. Cơ chế này giúp khi
        người dùng đổi mã chứng khoán, ứng dụng không bị treo chờ
        warmer đã chiếm token trước đó.
\end{enumerate}
