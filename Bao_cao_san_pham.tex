\chapter{Tổng quan đề tài}

\section{Tên đề tài}

\indent\textbf{FinScope -- Ứng dụng phân tích và dự báo chứng khoán
Việt Nam đa mô hình.}

\indent Sản phẩm là một ứng dụng web tương tác xây dựng trên nền tảng
Streamlit (Python), triển khai trực tuyến trên Streamlit Community
Cloud. FinScope tích hợp dữ liệu thực 53 mã chứng khoán niêm yết trên
Sở Giao dịch Chứng khoán Thành phố Hồ Chí Minh (HOSE) qua thư viện
\texttt{vnstock}, kết hợp 8 mô hình thống kê -- học máy cùng nhiều
công cụ phân tích kỹ thuật, tối ưu danh mục, đánh giá rủi ro và mô
phỏng giao dịch -- tạo nên một bộ công cụ học thuật toàn diện cho việc
nghiên cứu thị trường chứng khoán Việt Nam.

\section{Mô tả tổng quan sản phẩm}

\indent FinScope được tổ chức thành \textbf{14 trang chức năng độc
lập} chia sẻ chung một tầng dữ liệu và một bộ tham số người dùng
(mã chứng khoán, khoảng thời gian, tỉ lệ huấn luyện, ngôn ngữ
Việt -- Anh, chế độ sáng -- tối). Mọi trang đều bám theo cùng một
khuôn mẫu: tham số đầu vào ở thanh phía trên, bảng kết quả số ở giữa,
biểu đồ tương tác Plotly ở dưới và phần trích dẫn tài liệu tham khảo
cuối trang.

\subsection{Bốn nhóm tính năng cốt lõi}

\begin{itemize}[leftmargin=22pt,topsep=2pt,itemsep=4pt]
  \item \textbf{Nhóm 1 -- Phân tích thị trường và mã đơn lẻ}: Bảng
        điều khiển (Dashboard), Tổng quan thị trường (heatmap ngành,
        top tăng/giảm/khối lượng), Phân tích cơ bản (P/E, P/B, ROE,
        ROA, EPS, biên lợi nhuận), Phân tích kỹ thuật chi tiết
        (Ichimoku, MACD, RSI, Bollinger, Stochastic, ADX, OBV).
  \item \textbf{Nhóm 2 -- Dự báo và mô hình nâng cao}: tám mô hình dự
        báo (AR, MLR, ARIMA, SARIMA, ETS, GARCH, SARIMAX, GBR) cùng
        bộ gộp FinScope Ensemble; kiểm định khách quan
        Diebold--Mariano; biểu đồ quạt phân vị; Monte Carlo + công
        thức Itô.
  \item \textbf{Nhóm 3 -- Chiến lược và giao dịch}: bộ máy tín hiệu 8
        trụ cột chấm điểm tin cậy; ba kịch bản giao dịch Thận trọng /
        Cân bằng / Tích cực; Giao dịch Demo (Paper Trading) mô phỏng
        đặt lệnh với phí khớp lệnh 0,15\% và thuế bán 0,10\% theo
        biểu phí HOSE; nhật ký giao dịch (Thesis -- Lesson -- Tag --
        Rating); kiểm thử lùi (backtest) walk-forward.
  \item \textbf{Nhóm 4 -- Danh mục và cơ sở toán học}: tối ưu danh mục
        Markowitz (Equal Weight, Minimum-Variance, Tangency / Max
        Sharpe) kèm biên hiệu quả; mô hình CAPM với hệ số $\beta$,
        alpha Jensen và đường thị trường chứng khoán (SML); phân tích
        thành phần chính (PCA) với biểu đồ độ dốc và biplot; kiểm
        định đồng tích hợp Engle--Granger cho chiến lược giao dịch
        cặp; trang Cơ sở Toán học với 25 công thức LaTeX kèm trích dẫn
        APA 7th.
\end{itemize}

\subsection{Năng lực cá nhân hoá người dùng}

\begin{itemize}[leftmargin=22pt,topsep=2pt,itemsep=3pt]
  \item \textbf{Xác thực an toàn} bằng PBKDF2-HMAC-SHA256 200.000 vòng
        lặp; mật khẩu lưu dưới dạng băm + muối, không lưu mật khẩu
        gốc.
  \item \textbf{Dữ liệu cá nhân hoá} per-user: sổ Paper Trading,
        danh sách theo dõi (watchlist), cảnh báo giá, nhật ký giao
        dịch và huy hiệu thành tích (achievements).
  \item \textbf{Chế độ khách} cho phép truy cập nhanh không cần đăng
        ký, phù hợp khán giả hội thi và ban giám khảo trải nghiệm
        thử.
  \item \textbf{Song ngữ Việt -- Anh} và \textbf{chế độ sáng -- tối}
        toàn ứng dụng, chuyển đổi bằng nút trên thanh điều hướng.
\end{itemize}

\subsection{Đặc trưng kỹ thuật nổi bật}

\begin{itemize}[leftmargin=22pt,topsep=2pt,itemsep=3pt]
  \item \textbf{Hệ thống cache 5 tầng} (\texttt{cache\_data} đĩa 24
        giờ + 15 phút, \texttt{cache\_resource} singleton, hàng đợi
        khởi nóng nền và bộ chia tốc độ theo phiên) giúp chuyển trang
        dưới 1 giây sau khi đã ấm.
  \item \textbf{Bộ khởi nóng phân tầng} (tiered warmer) cho 53 mã
        HOSE, ưu tiên Hot / Warm / Cold, chạy ở 17,6 yêu cầu/phút
        (dưới mức 20 yêu cầu/phút của \texttt{vnstock}).
  \item \textbf{Khoảng 28.000 dòng Python} chia thành các tầng rõ
        ràng (\texttt{data}, \texttt{models}, \texttt{services},
        \texttt{charts}, \texttt{ui}, \texttt{auth}, \texttt{core},
        \texttt{app\_pages}) -- xem chi tiết tại Chương 3.
  \item \textbf{Triển khai trực tuyến} trên Streamlit Community Cloud,
        URL công khai
        \href{https://finscope-bfz4abyb4tq6fhg6rd3gkx.streamlit.app}{\texttt{finscope-bfz4abyb4tq6fhg6rd3gkx.streamlit.app}}
        (không cần cài đặt phía người dùng); mã nguồn cho phép chạy
        cục bộ phục vụ phát triển và kiểm thử.
\end{itemize}

\subsection{Đóng góp khoa học}

\begin{itemize}[leftmargin=22pt,topsep=2pt,itemsep=3pt]
  \item Tích hợp đồng thời tám mô hình dự báo và kiểm định
        Diebold--Mariano \emph{trong cùng một giao diện} cho dữ liệu
        HOSE -- điểm hiếm thấy ở các công cụ sinh viên hiện có.
  \item Trực quan hoá bốn thuật toán toán -- tài chính cổ điển
        (Markowitz 1952, CAPM 1964, PCA, Engle--Granger 1987) thành
        biểu đồ tương tác có thể chỉnh tham số trên trình duyệt.
  \item Cung cấp giao diện ``đọc được'' (explainable): mỗi tín hiệu
        gắn liền với điểm số trụ cột, mỗi dự báo gắn liền với chỉ số
        sai số (MAPE / RMSE / MAE / $R^2$), mỗi danh mục gắn liền với
        ràng buộc toán học và nguồn trích dẫn.
\end{itemize}

\section{Lý do và mục tiêu}

\indent Thị trường chứng khoán Việt Nam đang ngày càng thu hút sự quan
tâm của sinh viên ngành Toán -- Thống kê, song các công cụ phân tích
chuyên dụng phần lớn là phần mềm thương mại đắt tiền hoặc nền tảng nước
ngoài không hỗ trợ tốt cho mã HOSE. FinScope ra đời nhằm:

\begin{enumerate}[leftmargin=22pt,topsep=2pt,itemsep=4pt]
  \item Cung cấp môi trường \emph{học thuật miễn phí, mã nguồn mở} cho
   sinh viên thực hành các phương pháp dự báo chuỗi thời gian, phân
   tích kỹ thuật và lý thuyết danh mục trên dữ liệu thực Việt Nam.
  \item Trực quan hóa các \emph{thuật toán cổ điển} (Markowitz 1952,
   CAPM 1964, PCA, Engle--Granger 1987) bằng giao diện web tương
   tác, giúp người học hiểu trực giác trước khi đọc công thức.
  \item Triển khai 8 mô hình dự báo song song và \emph{kiểm định khách
   quan} (Diebold--Mariano) thay vì so sánh chủ quan -- nhấn mạnh tinh
   thần khoa học dữ liệu.
  \item Mô phỏng \emph{giao dịch ảo có phí thực tế HOSE} (phí khớp lệnh
   0{,}15\%, thuế bán 0{,}10\%) để người dùng luyện kỷ luật giao dịch
   mà không rủi ro vốn thật.
\end{enumerate}

\section{Phù hợp chủ đề ``Toán học trong kỷ nguyên trí tuệ nhân tạo''}

\indent Sản phẩm bám sát chủ đề hội thi qua ba khía cạnh: (i) ứng dụng
\emph{8 mô hình thống kê -- học máy} (AR, MLR, ARIMA, SARIMA, ETS, GARCH,
SARIMAX, Gradient Boosting) song song với \emph{kiểm định Diebold--Mariano}
khách quan để chọn mô hình tốt nhất theo dữ liệu, không dựa cảm tính;
(ii) tích hợp \emph{4 thuật toán toán -- tài chính cổ điển} (Markowitz,
CAPM, PCA, Engle--Granger Cointegration -- Nobel 2003) thường chỉ xuất
hiện trong giáo trình cao học, trực quan hoá thành biểu đồ tương tác;
(iii) sử dụng \emph{trợ lý AI lập trình} (mô hình ngôn ngữ lớn) làm
\textbf{công cụ hỗ trợ phát triển} -- nhóm vẫn giữ vai trò thiết kế, đọc
hiểu, kiểm thử và sửa lỗi mọi đoạn mã. Nhật ký sử dụng AI chi tiết được
trình bày trong phụ lục \texttt{Nhat\_ky\_su\_dung\_AI.pdf} theo đúng
quy chế hội thi.

trái phiếu, hợp đồng tương lai hay dữ liệu intraday.

\section{Tổng quan kiến trúc}

\indent FinScope được tổ chức theo \emph{kiến trúc phân tầng} (layered
architecture) gồm 5 tầng độc lập, mỗi tầng có nhiệm vụ rõ ràng và giao
tiếp qua hàm thuần (pure function) hoặc cache đĩa:

\begin{enumerate}[leftmargin=22pt,topsep=2pt,itemsep=4pt]
  \item \textbf{Tầng giao diện} (\texttt{app\_pages/}, \texttt{ui/}) --
   14 trang Streamlit + topbar, sidebar, CSS, components.
  \item \textbf{Tầng dịch vụ} (\texttt{services/}) -- logic nghiệp vụ
   thuần tính: signal engine 8 trụ, optimizer Markowitz, Monte Carlo,
   CAPM, PCA, cointegration, backtest, watchlist, alerts, journal.
  \item \textbf{Tầng mô hình} (\texttt{models/}) -- 8 mô hình thống kê
   -- học máy + Ensemble.
  \item \textbf{Tầng dữ liệu} (\texttt{data/}) -- truy xuất giá, tài
   chính, tin tức, paper trading book.
  \item \textbf{Tầng cấu hình \& tiện ích} (\texttt{core/},
   \texttt{auth/}) -- hằng số, theme, i18n, xác thực PBKDF2, cache,
   tham chiếu APA.
\end{enumerate}

\section{Tám mô hình dự báo}

\indent FinScope triển khai song song tám mô hình thống kê -- học máy
khác biệt về giả định và phạm vi áp dụng. Mỗi mô hình được trình bày
theo cùng một khuôn mẫu gồm bốn mục: \emph{ý tưởng -- công thức -- diễn
giải tham số -- tài liệu tham khảo}. Toàn bộ tám mô hình huấn luyện trên
cùng một tập dữ liệu (lịch sử đóng cửa hiệu chỉnh) và đánh giá bằng
cùng một bộ chỉ số (MAPE, RMSE, MAE, $R^2$).

\subsection{Mô hình tự hồi quy AR($p$) -- Autoregressive}

\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=4pt]
  \item \textbf{Ý tưởng:} giá trị hiện tại được giải thích bởi $p$ giá
        trị quá khứ kề nhau (mô hình tự hồi quy bậc $p$).
  \item \textbf{Công thức gốc:}
        \begin{equation}
          y_t \;=\; c \;+\; \sum_{i=1}^{p}\phi_i\, y_{t-i} \;+\; \varepsilon_t,
          \qquad \varepsilon_t \sim \mathcal{N}(0,\sigma^2).
          \label{eq:ar}
        \end{equation}
  \item \textbf{Diễn giải từng thành phần:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $y_t$: giá trị (giá đóng cửa hoặc lợi suất logarit) tại phiên $t$.
          \item $c$: hằng số chặn (intercept), ước lượng cùng với hệ số tự hồi quy.
          \item $\phi_i$: hệ số tự hồi quy bậc $i$, $i = 1,\dots,p$.
          \item $\varepsilon_t$: nhiễu trắng, kỳ vọng bằng 0, phương sai $\sigma^2$.
          \item $p$: bậc tự hồi quy, chọn theo Tiêu chuẩn Thông tin
                Akaike (AIC) \citep{Akaike1974} hoặc hàm tự tương quan
                riêng phần (PACF).
        \end{itemize}
  \item \textbf{Điều kiện và ước lượng:} chuỗi dừng yếu khi mọi nghiệm
        của $1 - \sum_{i=1}^{p}\phi_i z^i = 0$ nằm ngoài đường tròn
        đơn vị; ước lượng bằng phương pháp bình phương cực tiểu thông
        thường (OLS) hoặc phương trình Yule--Walker.
  \item \textbf{Tài liệu tham khảo:} \citet{BoxJenkins1970}, \citet{Akaike1974}.
\end{itemize}

\subsection{Hồi quy tuyến tính bội MLR -- Multiple Linear Regression}

\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=4pt]
  \item \textbf{Ý tưởng:} mô hình hoá biến mục tiêu bằng tổ hợp tuyến
        tính của $k$ biến ngoại sinh trễ (lợi suất logarit, biên độ
        biến động, khối lượng), $k = 3p$.
  \item \textbf{Công thức gốc:}
        \begin{equation}
          y_t \;=\; \beta_0 \;+\; \sum_{j=1}^{k}\beta_j\, x_{j,t}
          \;+\; \varepsilon_t.
          \label{eq:mlr}
        \end{equation}
  \item \textbf{Diễn giải từng thành phần:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $y_t$: biến phụ thuộc (giá hoặc lợi suất).
          \item $\beta_0$: hệ số chặn (intercept).
          \item $\beta_j$: hệ số hồi quy của biến độc lập thứ $j$.
          \item $x_{j,t}$: biến ngoại sinh thứ $j$ tại phiên $t$.
          \item $\varepsilon_t$: sai số ngẫu nhiên, giả định phân phối
                chuẩn $N(0,\sigma^2)$ độc lập đồng phân phối.
        \end{itemize}
  \item \textbf{Tính chất:} ước lượng bình phương cực tiểu thông thường
        (OLS) đạt tính chất ước lượng tuyến tính không chệch tốt nhất
        (BLUE -- Best Linear Unbiased Estimator) theo định lý
        Gauss--Markov khi các giả định tuyến tính, đồng nhất phương sai
        và độc lập được đáp ứng.
  \item \textbf{Tài liệu tham khảo:} \citet{Greene2018}.
\end{itemize}

\subsection{Mô hình tự hồi quy tích hợp trung bình trượt ARIMA($p,d,q$) -- Autoregressive Integrated Moving Average}

\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=4pt]
  \item \textbf{Ý tưởng:} kết hợp thành phần tự hồi quy bậc $p$, sai
        phân bậc $d$ và trung bình trượt bậc $q$ để xử lý chuỗi không
        dừng.
  \item \textbf{Công thức gốc:}
        \begin{equation}
          \Phi(L)\,(1-L)^{d}\, y_t \;=\; \Theta(L)\,\varepsilon_t.
          \label{eq:arima}
        \end{equation}
  \item \textbf{Diễn giải từng thành phần:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $L$: toán tử trễ (lag operator), $L\,y_t = y_{t-1}$.
          \item $\Phi(L) = 1 - \phi_1 L - \cdots - \phi_p L^p$: đa thức
                tự hồi quy (AR) bậc $p$.
          \item $\Theta(L) = 1 + \theta_1 L + \cdots + \theta_q L^q$: đa
                thức trung bình trượt (MA -- Moving Average) bậc $q$.
          \item $(1-L)^d$: toán tử sai phân bậc $d$, biến chuỗi không
                dừng thành dừng.
          \item $\varepsilon_t$: nhiễu trắng.
        \end{itemize}
  \item \textbf{Ước lượng:} cực đại hợp lý (MLE -- Maximum Likelihood
        Estimation); bậc $(p,d,q)$ chọn theo Tiêu chuẩn Thông tin Akaike
        (AIC) hoặc Tiêu chuẩn Thông tin Bayes (BIC).
  \item \textbf{Tài liệu tham khảo:} \citet{BoxJenkinsReinsel2008}.
\end{itemize}

\subsection{Mô hình ARIMA mùa vụ SARIMA($p,d,q$)($P,D,Q$)$_s$ -- Seasonal ARIMA}

\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=4pt]
  \item \textbf{Ý tưởng:} mở rộng Mô hình Tự hồi quy Tích hợp Trung
        bình trượt (ARIMA) với thành phần mùa vụ chu kỳ $s$ phiên.
        FinScope dùng $s = 5$ ứng với tuần giao dịch trên Sở Giao dịch
        Chứng khoán Thành phố Hồ Chí Minh (HOSE).
  \item \textbf{Công thức gốc:}
        \begin{equation}
          \Phi(L)\Phi_s(L^s)\,(1-L)^d(1-L^s)^D\, y_t
          \;=\; \Theta(L)\Theta_s(L^s)\,\varepsilon_t.
          \label{eq:sarima}
        \end{equation}
  \item \textbf{Diễn giải từng thành phần:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $\Phi(L),\,\Theta(L)$: đa thức tự hồi quy (AR) và trung
                bình trượt (MA) không mùa vụ.
          \item $\Phi_s(L^s),\,\Theta_s(L^s)$: đa thức tự hồi quy (AR)
                và trung bình trượt (MA) mùa vụ theo bước nhảy $s$.
          \item $(1-L^s)^D$: toán tử sai phân mùa vụ bậc $D$.
          \item $(P, D, Q)$: bậc tự hồi quy, sai phân và trung bình
                trượt mùa vụ.
          \item $s$: độ dài chu kỳ mùa vụ, $s = 5$ trong FinScope.
        \end{itemize}
  \item \textbf{Tài liệu tham khảo:} \citet{BoxJenkinsReinsel2008}.
\end{itemize}

\subsection{Mô hình làm mượt mũ Holt--Winters ETS -- Exponential Smoothing}

\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=4pt]
  \item \textbf{Ý tưởng:} phân rã chuỗi thời gian thành ba thành phần
        \emph{mức (level)}, \emph{xu thế (trend)} và \emph{mùa vụ
        (seasonal)}; mỗi thành phần làm mượt theo cấp số nhân.
  \item \textbf{Công thức gốc (mô hình cộng):}
        \begin{align}
          \ell_t &= \alpha (y_t - s_{t-m}) + (1-\alpha)(\ell_{t-1} + b_{t-1}),\\
          b_t    &= \beta (\ell_t - \ell_{t-1}) + (1-\beta) b_{t-1},
                   \label{eq:ets}\\
          s_t    &= \gamma (y_t - \ell_{t-1} - b_{t-1}) + (1-\gamma) s_{t-m}.
        \end{align}
  \item \textbf{Diễn giải từng thành phần:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $\ell_t$: thành phần mức (level), giá trị trung bình hiện tại.
          \item $b_t$: thành phần xu thế (trend), độ dốc tăng/giảm.
          \item $s_t$: thành phần mùa vụ (seasonal) chu kỳ $m$.
          \item $\alpha,\beta,\gamma \in [0,1]$: ba tham số làm mượt,
                ước lượng cực tiểu Tổng bình phương sai số (SSE -- Sum
                of Squared Errors) trên tập huấn luyện.
          \item $m$: độ dài chu kỳ mùa vụ.
        \end{itemize}
  \item \textbf{Tài liệu tham khảo:} \citet{Holt1957}, \citet{Winters1960};
        tổng hợp hiện đại: \citet{HyndmanAthanasopoulos2021}.
\end{itemize}

\subsection{Mô hình phương sai có điều kiện tổng quát GARCH(1,1) -- Generalized Autoregressive Conditional Heteroskedasticity}

\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=4pt]
  \item \textbf{Ý tưởng:} mô hình hoá \emph{phương sai có điều kiện}
        của lợi suất logarit, nắm bắt hiện tượng cụm biến động
        (volatility clustering).
  \item \textbf{Công thức gốc:}
        \begin{equation}
          \sigma_t^2 \;=\; \omega \;+\; \alpha\,\varepsilon_{t-1}^{2}
                                  \;+\; \beta\,\sigma_{t-1}^{2}.
          \label{eq:garch}
        \end{equation}
  \item \textbf{Diễn giải từng thành phần:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $\sigma_t^2$: phương sai có điều kiện tại phiên $t$.
          \item $\omega > 0$: hằng số đảm bảo phương sai dương ngặt.
          \item $\alpha \ge 0$: hệ số tự hồi quy có điều kiện
                (ARCH -- Autoregressive Conditional Heteroskedasticity),
                đo phản ứng theo cú sốc gần nhất.
          \item $\beta \ge 0$: hệ số tổng quát hoá (GARCH), đo tính
                dai dẳng của biến động.
          \item Điều kiện dừng: $\alpha + \beta < 1$.
          \item $\varepsilon_{t-1}^{2}$: bình phương sai số phiên trước.
        \end{itemize}
  \item \textbf{Tài liệu tham khảo:} \citet{Bollerslev1986}, mở rộng từ
        mô hình ARCH của \citet{Engle1982}.
\end{itemize}

\subsection{Mô hình SARIMAX -- SARIMA with Exogenous Regressors}

\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=4pt]
  \item \textbf{Ý tưởng:} mở rộng mô hình SARIMA cho biến ngoại sinh
        $\mathbf{x}_t$; trong FinScope là khối lượng (Volume) và biên
        độ trong ngày (Range).
  \item \textbf{Công thức gốc:}
        \begin{equation}
          \Phi(L)\Phi_s(L^s)\,(1-L)^d(1-L^s)^D\bigl(y_t -
          \boldsymbol{\beta}^{\!\top}\mathbf{x}_t\bigr)
          \;=\; \Theta(L)\Theta_s(L^s)\,\varepsilon_t.
          \label{eq:sarimax}
        \end{equation}
  \item \textbf{Diễn giải từng thành phần:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $\mathbf{x}_t$: vec-tơ biến ngoại sinh tại phiên $t$.
          \item $\boldsymbol{\beta}$: vec-tơ hệ số hồi quy cho biến
                ngoại sinh.
          \item Các thành phần còn lại giữ nguyên ý nghĩa của Mô hình
                SARIMA (xem mục 4.2.4).
        \end{itemize}
  \item \textbf{Cài đặt:} biểu diễn không gian trạng thái (state-space
        representation) theo \citet{Hyndman2008state}.
  \item \textbf{Tài liệu tham khảo:} \citet{BoxJenkinsReinsel2008},
        \citet{Hyndman2008state}.
\end{itemize}

\subsection{Hồi quy tăng cường gradient GBR -- Gradient Boosting Regression}

\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=4pt]
  \item \textbf{Ý tưởng:} mô hình tổ hợp (ensemble) cộng dồn các cây hồi
        quy yếu; mỗi cây được học để khớp với gradient âm của hàm mất
        mát từ bộ tổ hợp trước -- nguyên lý tăng cường (boosting) của
        Friedman.
  \item \textbf{Công thức gốc:}
        \begin{equation}
          F_M(\mathbf{x}) \;=\; F_0(\mathbf{x}) \;+\;
          \sum_{m=1}^{M} \nu\, h_m(\mathbf{x}),
          \qquad
          h_m \;=\; \arg\min_{h}\sum_{i=1}^{n}
                    \bigl[\,r_{i,m} - h(\mathbf{x}_i)\,\bigr]^2.
          \label{eq:gbr}
        \end{equation}
  \item \textbf{Diễn giải từng thành phần:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $F_M(\mathbf{x})$: dự báo cuối cùng sau $M$ vòng tăng cường (boosting).
          \item $F_0(\mathbf{x})$: giá trị khởi tạo, thường là trung bình mẫu.
          \item $h_m(\mathbf{x})$: cây hồi quy yếu thứ $m$.
          \item $r_{i,m} = -\partial L(y_i,F_{m-1}(\mathbf{x}_i))/\partial F_{m-1}$:
                gradient âm, còn gọi là phần dư giả (pseudo-residual).
          \item $\nu \in (0,1]$: tốc độ học (learning rate), co rút
                ảnh hưởng của mỗi cây.
          \item $M$: số vòng tăng cường (boosting iterations).
        \end{itemize}
  \item \textbf{Tài liệu tham khảo:} \citet{Friedman2001}.
\end{itemize}

\subsection{Bộ gộp FinScope Ensemble}

\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=4pt]
  \item \textbf{Ý tưởng:} gộp tám dự báo bằng \emph{trọng số nghịch đảo
        sai số}; mô hình chính xác hơn nhận trọng số lớn hơn.
  \item \textbf{Công thức gốc:}
        \begin{equation}
          \hat{y}_t^{\,\text{ens}} \;=\; \sum_{i=1}^{8} w_i\,\hat{y}_t^{(i)},
          \qquad
          w_i \;=\; \dfrac{1/(\text{MAPE}_i + 0{,}1)}
                          {\sum_{j=1}^{8} 1/(\text{MAPE}_j + 0{,}1)}.
          \label{eq:ensemble}
        \end{equation}
  \item \textbf{Diễn giải từng thành phần:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $\hat{y}_t^{(i)}$: dự báo của mô hình thứ $i$ tại phiên $t$.
          \item $w_i$: trọng số gộp, tổng các $w_i$ bằng 1.
          \item $\text{MAPE}_i$: Sai số Phần trăm Tuyệt đối Trung bình
                (Mean Absolute Percentage Error) của mô hình $i$ trên
                tập kiểm thử.
          \item $0{,}1$: hằng số làm trơn, chống chia cho 0 khi MAPE quá nhỏ.
        \end{itemize}
  \item \textbf{Cơ sở lý thuyết:} \citet{StockWatson2004} chứng minh
        thực nghiệm phương pháp gộp thường vượt mô hình tốt nhất riêng
        lẻ; nền tảng lý thuyết kinh điển: \citet{BatesGranger1969}.
\end{itemize}

\section{Kiểm định Diebold--Mariano (DM, 1995)}

\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=4pt]
  \item \textbf{Ý tưởng:} so sánh khách quan độ chính xác giữa hai mô
        hình dự báo dựa trên thống kê tiệm cận phân phối chuẩn của
        trung bình chênh tổn thất.
  \item \textbf{Công thức gốc} (hiệu chỉnh mẫu nhỏ
        Harvey--Leybourne--Newbold, HLN 1997):
        \begin{equation}
          \text{DM} = \frac{\bar{d}}{\sqrt{\hat{V}(\bar{d})/n}} \;\to\; N(0,1),
          \qquad d_t = L(e_{1,t}) - L(e_{2,t}).
          \label{eq:dm}
        \end{equation}
  \item \textbf{Diễn giải từng thành phần:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $e_{1,t},\,e_{2,t}$: sai số dự báo của mô hình 1 và 2
                tại phiên $t$.
          \item $L(\cdot)$: hàm tổn thất (loss function), thường lấy
                $L(e) = e^2$ (bình phương).
          \item $d_t$: chênh lệch tổn thất giữa hai mô hình tại phiên $t$.
          \item $\bar{d}$: trung bình mẫu của $d_t$.
          \item $\hat{V}(\bar{d})$: ước lượng phương sai dài hạn (long-run
                variance) của $\bar{d}$, có hiệu chỉnh tự tương quan.
          \item $n$: số quan sát kiểm thử.
          \item $N(0,1)$: phân phối tiệm cận chuẩn hoá; bác bỏ giả thuyết
                hai mô hình tương đương khi $|\mathrm{DM}|$ lớn.
        \end{itemize}
  \item \textbf{Ứng dụng:} áp dụng cho mọi cặp $(i, j)$ trong 8 mô hình
        và đối chứng với Bước đi ngẫu nhiên (Random Walk), in ra ma trận
        giá trị $p$ một cách trực quan.
  \item \textbf{Tài liệu tham khảo:} \citet{DieboldMariano1995},
        \citet{HarveyLeybourneNewbold1997}.
\end{itemize}

\section{Phân tích Ichimoku Kinko Hyo (Hosoda 1969)}

\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=4pt]
  \item \textbf{Ý tưởng:} hệ thống biểu đồ ``một cái nhìn cân bằng''
        gồm năm thành phần đo trên cửa sổ 9, 26 và 52 phiên giúp xác
        định xu hướng, hỗ trợ và kháng cự đồng thời.
  \item \textbf{Công thức gốc:}
        \begin{align}
          \text{Tenkan-sen} &= \tfrac{1}{2}\!\left(\max_{9}H + \min_{9}L\right),\\
          \text{Kijun-sen}  &= \tfrac{1}{2}\!\left(\max_{26}H + \min_{26}L\right),
                              \label{eq:ichimoku}\\
          \text{Senkou A}_t &= \tfrac{1}{2}\!\left(\text{Tenkan}_t + \text{Kijun}_t\right) \;\text{(đẩy +26)},\\
          \text{Senkou B}_t &= \tfrac{1}{2}\!\left(\max_{52}H + \min_{52}L\right) \;\text{(đẩy +26)},\\
          \text{Chikou}_t   &= C_t \;\text{(đẩy }-26).
        \end{align}
  \item \textbf{Diễn giải từng thành phần:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item Tenkan-sen (đường chuyển đổi): trung bình cao--thấp 9 phiên.
          \item Kijun-sen (đường cơ sở): trung bình cao--thấp 26 phiên.
          \item Senkou Span A và B: tạo ``mây'' (Kumo) chiếu trước 26 phiên.
          \item Chikou Span (đường trễ): giá đóng cửa lùi về 26 phiên.
          \item $H_t, L_t, C_t$: giá cao nhất, thấp nhất, đóng cửa tại phiên $t$.
        \end{itemize}
  \item \textbf{Phân tầng tín hiệu:} sản phẩm chia tín hiệu thành bốn
        tầng -- vị trí so với mây, giao cắt Tenkan--Kijun (TK cross),
        xác nhận của Chikou và mây tương lai. Toàn bộ tuân thủ nguyên
        tắc \emph{anti-leak}: tại thời điểm $t$ chỉ dùng dữ liệu có sẵn
        đến $t$.
  \item \textbf{Tài liệu tham khảo:} \citet{Hosoda1969}.
\end{itemize}

\section{Lý thuyết danh mục Markowitz (1952)}

\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=4pt]
  \item \textbf{Ý tưởng:} chọn vec-tơ trọng số $\mathbf{w}$ cho danh mục
        chỉ mua (long-only) để cực tiểu rủi ro với một mức lợi suất kỳ
        vọng cho trước, hình thành biên hiệu quả (efficient frontier).
  \item \textbf{Công thức gốc (bài toán quy hoạch toàn phương):}
        \begin{equation}
          \min_{\mathbf{w}}\;\mathbf{w}^{\!\top}\boldsymbol{\Sigma}\mathbf{w}
          \quad\text{s.t.}\quad
          \mathbf{w}^{\!\top}\boldsymbol{\mu} = R_{\text{target}},\;\;
          \mathbf{w}^{\!\top}\mathbf{1} = 1,\;\;
          \mathbf{w}\ge\mathbf{0}.
          \label{eq:markowitz}
        \end{equation}
  \item \textbf{Diễn giải từng thành phần:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $\mathbf{w} \in \mathbb{R}^n$: vec-tơ trọng số $n$ tài sản.
          \item $\boldsymbol{\Sigma}$: ma trận hiệp phương sai lợi suất
                hoá năm, kích thước $n\times n$.
          \item $\boldsymbol{\mu}$: vec-tơ lợi suất kỳ vọng hoá năm.
          \item $R_{\text{target}}$: mức lợi suất kỳ vọng mục tiêu.
          \item $\mathbf{w}^{\!\top}\mathbf{1} = 1$: ràng buộc tổng trọng số.
          \item $\mathbf{w}\ge\mathbf{0}$: ràng buộc chỉ mua (long-only).
        \end{itemize}
  \item \textbf{Ba danh mục đặc biệt và nghiệm đóng:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item Equal-Weight: $w_i = 1/n$ (chuẩn so sánh -- benchmark).
          \item Minimum-Variance:
                $\mathbf{w}^{\star}_{\text{MV}} \propto \boldsymbol{\Sigma}^{-1}\mathbf{1}$.
          \item Tangency (Max Sharpe):
                $\mathbf{w}^{\star}_{\text{tan}} \propto
                 \boldsymbol{\Sigma}^{-1}(\boldsymbol{\mu} - r_f\mathbf{1})$.
        \end{itemize}
  \item \textbf{Cài đặt biên hiệu quả:} hạ gradient có chiếu (projected
        gradient descent) thuần \texttt{numpy}, kết hợp chiếu lên đơn
        hình (simplex) theo \citet{WangCarreira2013}.
  \item \textbf{Tài liệu tham khảo:} \citet{Markowitz1952},
        \citet{WangCarreira2013}.
\end{itemize}

\section{Mô hình CAPM -- Capital Asset Pricing Model (Sharpe 1964, Lintner 1965)}

\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=4pt]
  \item \textbf{Ý tưởng:} lợi suất kỳ vọng vượt phi rủi ro của một mã
        tỉ lệ với phần bù rủi ro thị trường, hệ số tỉ lệ chính là beta.
  \item \textbf{Công thức gốc:}
        \begin{equation}
          E[R_i] - r_f \;=\; \beta_i \bigl(E[R_m] - r_f\bigr),
          \qquad
          \beta_i \;=\; \frac{\mathrm{Cov}(R_i, R_m)}{\mathrm{Var}(R_m)}.
          \label{eq:capm}
        \end{equation}
  \item \textbf{Hệ số alpha Jensen} (Jensen 1968) đo lợi suất vượt CAPM:
        \begin{equation}
          \alpha_i \;=\; E[R_i] - r_f - \beta_i \bigl(E[R_m] - r_f\bigr).
          \label{eq:jensen}
        \end{equation}
  \item \textbf{Diễn giải từng thành phần:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $E[R_i]$: lợi suất kỳ vọng của mã $i$.
          \item $E[R_m]$: lợi suất kỳ vọng của chỉ số thị trường (FinScope
                dùng VN-Index).
          \item $r_f$: lãi suất phi rủi ro (mặc định 4,7\%/năm theo lãi
                gửi 12 tháng Ngân hàng Nhà nước Việt Nam -- SBV).
          \item $\beta_i$: hệ số nhạy cảm thị trường của mã $i$.
          \item $\mathrm{Cov}(R_i, R_m)$: hiệp phương sai lợi suất.
          \item $\mathrm{Var}(R_m)$: phương sai lợi suất thị trường.
          \item $\alpha_i$: lợi suất vượt CAPM của mã $i$; $\alpha_i > 0$
                được coi là vượt trội thị trường (outperform) sau hiệu
                chỉnh rủi ro.
        \end{itemize}
  \item \textbf{Cài đặt:} hồi quy bình phương cực tiểu thông thường
        (OLS) lợi suất từng mã trên lợi suất VN-Index; tính giá trị
        $p$ hai phía qua hàm sống sót của phân phối $t$ Student
        (\texttt{scipy.stats.t.sf}) để giữ độ chính xác khi $|t|$ lớn.
  \item \textbf{Tài liệu tham khảo:} \citet{Sharpe1964}, \citet{Lintner1965}.
\end{itemize}

\section{Phân tích thành phần chính PCA -- Principal Component Analysis (Hotelling 1933)}

\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=4pt]
  \item \textbf{Ý tưởng:} chiếu dữ liệu lợi suất nhiều chiều xuống các
        trục mới (thành phần chính) sao cho phương sai giải thích lớn
        nhất, giúp nhận diện yếu tố ẩn (thị trường, ngành, quy mô).
  \item \textbf{Công thức gốc} (phân rã trị riêng):
        \begin{equation}
          \boldsymbol{\Sigma} \;=\; \mathbf{V}\boldsymbol{\Lambda}\mathbf{V}^{\!\top},
          \qquad
          \text{phương sai giải thích}_i \;=\;
          \dfrac{\lambda_i}{\sum_{j=1}^{n}\lambda_j}.
          \label{eq:pca}
        \end{equation}
  \item \textbf{Diễn giải từng thành phần:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $\boldsymbol{\Sigma}$: ma trận tương quan (hoặc hiệp phương
                sai) lợi suất danh mục.
          \item $\mathbf{V}$: ma trận vec-tơ riêng, mỗi cột là một thành
                phần chính (Principal Component -- PC).
          \item $\boldsymbol{\Lambda}$: ma trận đường chéo trị riêng
                $\lambda_1 \ge \lambda_2 \ge \cdots \ge \lambda_n$.
          \item $\lambda_i$: phương sai dọc theo thành phần chính thứ $i$.
        \end{itemize}
  \item \textbf{Diễn giải tài chính:} thành phần chính thứ nhất (PC1)
        thường là yếu tố thị trường (market factor); PC2 và PC3 có thể
        là yếu tố ngành (sector), quy mô (size) hoặc đà giá (momentum).
        Sản phẩm hiển thị biểu đồ độ dốc (scree plot) và biểu đồ kép
        (biplot) PC1 và PC2.
  \item \textbf{Tài liệu tham khảo:} \citet{Hotelling1933}.
\end{itemize}

\section{Đồng tích hợp Engle--Granger (1987) -- Nobel 2003}

\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=4pt]
  \item \textbf{Ý tưởng:} hai chuỗi không dừng vẫn có thể có tổ hợp
        tuyến tính dừng -- gọi là đồng tích hợp; chênh giá hai mã đồng
        tích hợp có tính phục hồi trung bình (mean-reverting) thích
        hợp giao dịch cặp (pairs trading).
  \item \textbf{Công thức gốc (kiểm định hai bước):}
        \begin{align}
          \text{Bước 1:}\;\;& Y_t \;=\; \alpha + \beta X_t + u_t
                              \quad \text{(OLS).} \label{eq:eg1}\\
          \text{Bước 2:}\;\;& \Delta u_t \;=\; \rho\, u_{t-1}
                              + \sum_{i=1}^{p}\gamma_i \Delta u_{t-i}
                              + \varepsilon_t,\;\; H_0:\,\rho = 0.
                              \label{eq:eg2}
        \end{align}
  \item \textbf{Diễn giải từng thành phần:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $X_t, Y_t$: hai chuỗi giá ban đầu (cùng đơn vị).
          \item $\alpha, \beta$: hệ số chặn và hệ số đồng tích hợp.
          \item $u_t$: phần dư hồi quy bước 1 (chênh chuỗi -- spread).
          \item $\Delta u_t = u_t - u_{t-1}$: sai phân bậc một của $u_t$.
          \item $\rho$: hệ số trong kiểm định Dickey--Fuller mở rộng
                (ADF -- Augmented Dickey--Fuller).
          \item $\gamma_i$: hệ số trễ trong ADF, $i = 1,\dots,p$.
          \item $H_0:\;\rho = 0$: giả thuyết không, chuỗi $u_t$ có
                nghiệm đơn vị (không dừng).
        \end{itemize}
  \item \textbf{Kết luận:} bác bỏ $H_0$ với $p < 0{,}05$ tương đương kết
        luận hai chuỗi đồng tích hợp. Sản phẩm in bảng các cặp đồng
        tích hợp tốt nhất kèm điểm chuẩn hoá ($z$-score) của spread
        hiện tại trong dải $\pm 2\sigma$ để người dùng nhận diện cơ hội.
  \item \textbf{Tài liệu tham khảo:} \citet{EngleGranger1987}.
\end{itemize}

\section{Mô phỏng Monte Carlo, công thức Itô và VaR/CVaR}

\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=4pt]
  \item \textbf{Ý tưởng:} mô phỏng vốn danh mục $N$ ngày tới bằng hai
        phương pháp -- tái lấy mẫu lịch sử (bootstrap) và chuyển động
        Brown hình học tham số (GBM -- Geometric Brownian Motion) --
        rồi tính các phân vị, xác suất sinh lời và rủi ro đuôi VaR/CVaR.
  \item \textbf{Công thức GBM với hiệu chỉnh Itô:}
        \begin{equation}
          d \log S_t \;=\; \Bigl(\mu - \tfrac{\sigma^2}{2}\Bigr) dt
                          + \sigma\, dW_t.
          \label{eq:gbm}
        \end{equation}
  \item \textbf{Diễn giải từng thành phần:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $S_t$: giá tài sản tại thời điểm $t$.
          \item $\mu$: tốc độ tăng trưởng kỳ vọng (drift) hoá năm.
          \item $\sigma$: độ biến động (volatility) hoá năm.
          \item $\sigma^2/2$: hiệu chỉnh Itô, đảm bảo
                $E[S_T] = S_0\, e^{\mu T}$ đúng theo lý thuyết trung
                lập rủi ro (risk-neutral).
          \item $W_t$: chuyển động Brown chuẩn (standard Brownian motion).
        \end{itemize}
  \item \textbf{Giá trị rủi ro (VaR) và giá trị rủi ro có điều kiện
        (CVaR):}
        \begin{equation}
          \mathrm{VaR}_\alpha \;=\; -F_R^{-1}(\alpha),
          \qquad
          \mathrm{CVaR}_\alpha \;=\;
          -\,E\!\left[R \;\big|\; R \le F_R^{-1}(\alpha)\right].
          \label{eq:varcvar}
        \end{equation}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $\alpha$: mức tin cậy đuôi (mặc định 5\%).
          \item $R$: biến lợi suất ngẫu nhiên.
          \item $F_R$: hàm phân phối tích luỹ (CDF) của $R$.
          \item $F_R^{-1}(\alpha)$: phân vị $\alpha$ của $R$.
          \item $\mathrm{VaR}_\alpha$: mức lỗ tối đa với độ tin cậy $1-\alpha$.
          \item $\mathrm{CVaR}_\alpha$: mức lỗ kỳ vọng \emph{khi đã vượt} VaR.
        \end{itemize}
  \item \textbf{Tài liệu tham khảo:} \citet{Jorion2007},
        \citet{RockafellarUryasev2000}.
\end{itemize}

\section{Quản trị rủi ro}

\indent FinScope sử dụng năm chỉ số rủi ro chuẩn ngành, mỗi chỉ số
trình bày theo cùng khuôn mẫu công thức kèm diễn giải.

\subsection{Khoảng biến động trung bình (ATR -- Average True Range)}
\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=3pt]
  \item \textbf{Công thức gốc:}
        \begin{equation}
          \mathrm{TR}_t = \max\!\bigl(H_t - L_t,\; |H_t - C_{t-1}|,\;
                                       |L_t - C_{t-1}|\bigr),\quad
          \mathrm{ATR}_t = \frac{(n-1)\,\mathrm{ATR}_{t-1} + \mathrm{TR}_t}{n}.
          \label{eq:atr}
        \end{equation}
  \item \textbf{Diễn giải:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $H_t, L_t, C_t$: giá cao nhất, thấp nhất, đóng cửa tại phiên $t$.
          \item $\mathrm{TR}_t$: khoảng biến động thực (True Range) tại phiên $t$.
          \item $n$: cửa sổ tính trung bình (mặc định 14 phiên).
        \end{itemize}
  \item \textbf{Ứng dụng:} cắt lỗ theo bội số ATR (Wilder 1978).
\end{itemize}

\subsection{Tỉ số Sharpe (Sharpe ratio)}
\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=3pt]
  \item \textbf{Công thức gốc:}
        \begin{equation}
          \mathrm{Sharpe} \;=\; \frac{E[R_p] - r_f}{\sigma_p} \cdot \sqrt{252}.
          \label{eq:sharpe}
        \end{equation}
  \item \textbf{Diễn giải:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $E[R_p]$: lợi suất kỳ vọng hằng ngày của danh mục.
          \item $\sigma_p$: độ lệch chuẩn lợi suất ngày (ước lượng không
                thiên, $\mathrm{ddof} = 1$).
          \item $r_f$: lãi suất phi rủi ro (mặc định 4,7\%/năm).
          \item $\sqrt{252}$: hệ số chuẩn hoá năm (252 phiên giao dịch).
        \end{itemize}
  \item \textbf{Tài liệu tham khảo:} \citet{Sharpe1966}.
\end{itemize}

\subsection{Tỉ số Sortino (Sortino ratio)}
\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=3pt]
  \item \textbf{Công thức gốc:}
        \begin{equation}
          \mathrm{Sortino} = \frac{E[R_p] - r_f}{\sigma_p^{-}}\cdot \sqrt{252},
          \quad
          \sigma_p^{-} = \sqrt{E\!\left[\min(R_p - r_f,\,0)^2\right]}.
          \label{eq:sortino}
        \end{equation}
  \item \textbf{Diễn giải:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $\sigma_p^{-}$: độ lệch chuẩn lợi suất phía âm (downside
                deviation), chỉ tính các quan sát thua so với $r_f$.
          \item Phù hợp khi phân phối lợi suất bất đối xứng -- chỉ phạt
                biến động bất lợi.
        \end{itemize}
  \item \textbf{Tài liệu tham khảo:} \citet{SortinoPrice1994}.
\end{itemize}

\subsection{Mức rớt sâu nhất (MDD -- Maximum Drawdown)}
\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=3pt]
  \item \textbf{Công thức gốc:}
        \begin{equation}
          \mathrm{MDD} \;=\; \min_{t}\;
          \frac{E_t - \max_{s \le t} E_s}{\max_{s \le t} E_s}.
          \label{eq:mdd}
        \end{equation}
  \item \textbf{Diễn giải:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $E_t$: giá trị vốn (equity) tại phiên $t$.
          \item $\max_{s \le t} E_s$: đỉnh vốn cao nhất từ đầu đến phiên $t$.
          \item Giá trị MDD càng gần 0 càng an toàn (luôn $\le 0$).
        \end{itemize}
  \item \textbf{Tài liệu tham khảo:} \citet{MagdonIsmail2004}.
\end{itemize}

\subsection{Tiêu chí Kelly (Kelly criterion)}
\begin{itemize}[leftmargin=18pt,topsep=2pt,itemsep=3pt]
  \item \textbf{Công thức gốc:}
        \begin{equation}
          f^{\star} = W - \frac{1 - W}{b},\quad
          b = \frac{\text{thắng trung bình}}{|\text{lỗ trung bình}|},\quad
          g^{\star} = W \log(1 + f^{\star}b) + (1 - W)\log(1 - f^{\star}).
          \label{eq:kelly}
        \end{equation}
  \item \textbf{Diễn giải:}
        \begin{itemize}[leftmargin=14pt,topsep=2pt,itemsep=1pt]
          \item $W$: tỉ lệ thắng (win-rate) trong lịch sử lệnh.
          \item $b$: tỉ số kỳ vọng thắng trên lỗ (payoff ratio).
          \item $f^{\star}$: tỉ lệ vốn tối ưu (full Kelly) mỗi lệnh.
          \item $g^{\star}$: tốc độ tăng trưởng kỳ vọng log-vốn.
          \item Sản phẩm dùng 1/2 Kelly và 1/4 Kelly để giảm phương sai
                (Thorp 1969).
        \end{itemize}
  \item \textbf{Tài liệu tham khảo:} \citet{Kelly1956}, \citet{Thorp1969}.
\end{itemize}
